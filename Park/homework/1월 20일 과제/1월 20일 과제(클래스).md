[User.java]

```java
public class User {

	String id;
	int password;
	String name;
	String email;
	int age;

	public User() {

	}

	public User(String id, int password, String name, String email, int age) {
		this.id = id;
		this.password = password;
		this.name = name;
		this.email = email;
		this.age = age;
	}

	@Override
	public String toString() {
		return "User [id=" + id + ", password=" + password + ", name=" + name + ", email=" + email + ", age=" + age
				+ "]";
	}

}
```
