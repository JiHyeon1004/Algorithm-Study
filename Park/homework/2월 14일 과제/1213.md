import java.util.Scanner;

public class Problem1213 {

	// 1213. String
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 10번의 테스트 케이스를 실행하는 반복문
		for (int i = 0; i < 10; i++) {

			// 특정 문자열의 개수를 저장할 변수
			int count = 0;

			// 테스트 케이스 번호 입력
			int testCaseNum = scan.nextInt();
			// 찾을 문자열 입력
			String pattern = scan.next();
			// 주어진 문자열 입력
			String text = scan.next();
			// 주어진 문자열을 배열로 변환
			String[] textArr = text.split("");
			// 찾을 문자열을 배열로 변환
			String[] patternArr = pattern.split("");
			// 비교할 원문 문자열 인덱스
			int textIndex = 0;
			// 비교할 패턴의 인덱스
			int patternIndex = 0;

			// 원문의 마지막 글자까지 검색하는 반복문
			while (textIndex < textArr.length) {

				// 글자가 다르면
				// 출발점을 원점으로 이동하고
				// 패턴의 출발점을 원점 전칸으로 이동하고
				if (!(textArr[textIndex].equals(patternArr[patternIndex]))) {
					textIndex = textIndex - patternIndex;
					patternIndex = -1;
				}

				// 패턴의 길이까지 다 같으면 개수를 추가하고
				// 다시 시작점을 원점으로
				if (patternIndex == patternArr.length - 1) {
					count++;
					patternIndex = -1;
				}

				// 출발점을 한 칸 이동
				textIndex++;
				// 비교할 패턴의 위치 한 칸 이동
				patternIndex++;

			}

			// 결과 출력
			System.out.println("#" + testCaseNum + " " + count);

		}

		scan.close();

	}

}
