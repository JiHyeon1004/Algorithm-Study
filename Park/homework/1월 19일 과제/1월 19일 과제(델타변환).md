public static void main(String[] args) {

		Scanner scan = new Scanner(System.in);

		System.out.print("지형 범위를 입력하세요. >> ");

		int size = scan.nextInt();

		int[][] place = new int[size][size];

		System.out.println("지형 정보를 입력하세요.");

		for (int i = 0; i < size; i++) {
			for (int j = 0; j < size; j++) {
				place[i][j] = scan.nextInt();
			}
		}

		// 상하좌우 4방향
		int[] dr = { -1, 0, 1, 0 };
		int[] dc = { 0, 1, 0, -1 };

		int[][] distance = new int[size * size][4];

		for (int i = 0; i < size; i++) {
			for (int j = 0; j < size; j++) {
				int r = i;
				int c = j;

				if (place[r][c] == 0) {
					continue;
				}

				for (int k = 0; k < 4; k++) {
					int nr = r + dr[k];
					int nc = c + dc[k];

					if (nr < 0 || nr >= size || nc < 0 || nc >= size) {
						continue;
					}

					int count = 0;

					for (int l = 0; l < size; l++) {
						if (place[nr][nc] == 0) {
							count++;
							nr = nr + dr[k];
							nc = nc + dc[k];

							if (nr < 0 || nr >= size || nc < 0 || nc >= size) {
								break;
							}
						}
					}
					distance[size * i + j][k] = count + 1;
				}
			}
		}

		for (int i = 0; i < size * size; i++) {
			for (int j = 0; j < 4; j++) {
				System.out.print(distance[i][j] + "\t");
			}
			System.out.println("");
		}

		int max = 1;

		for (int i = 0; i < size * size; i++) {
			for (int j = 0; j < 4; j++) {
				if (max < distance[i][j]) {
					max = distance[i][j];
				}
			}
		}

		System.out.println(max);

		scan.close();

	}
