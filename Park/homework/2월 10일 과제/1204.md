```java
import java.util.Scanner;

public class Problem1204 {

	// 1204. 최빈수 구하기
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 테스트 케이스 크기 입력
		int testCase = scan.nextInt();
		// 테스트 케이스별 결과를 저장할 배열 생성
		int[] resultSet = new int[testCase];

		// 테스트 케이스별 결과를 계산하는 반복문
		for (int test = 0; test < testCase; test++) {
			// 테스트 케이스의 번호 입력
			int testCaseNum = scan.nextInt() - 1;
			// 점수별 빈도를 저장할 배열
			int[] score = new int[101];

			// 입력한 점수에 해당하는 인덱스에 1 추가하여 빈도수 추가
			for (int i = 0; i < 1000; i++) {
				// 입력한 학생 점수
				int studentScore = scan.nextInt();

				// 해당 점수의 빈도 1 추가
				score[studentScore]++;
			}

			// 최빈 점수의 위치를 저장할 변수
			int index = 0;
			// 최빈 점수의 개수
			int maxCommonScore = 0;

			// 최빈 점수의 위치를 구하는 반복문
			for (int i = 0; i < 101; i++) {
				if (maxCommonScore <= score[i]) {
					maxCommonScore = score[i];
					index = i;
				}
			}

			// 인덱스가 점수이므로
			// 인덱스를 결과 배열에 저장
			resultSet[testCaseNum] = index;
		}

		// 출력 양식에 맞춰 출력
		for (int i = 0; i < testCase; i++) {
			System.out.println("#" + (i + 1) + " " + resultSet[i]);
		}

		// Scanner 종료
		scan.close();

	}

}
```
