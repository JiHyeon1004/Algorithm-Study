import java.util.Arrays;
import java.util.Scanner;

public class Problem1209 {

	// 1209. Sum
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 테스트 케이스별 결과를 출력하는 반복문
		for (int testCase = 0; testCase < 10; testCase++) {

			// 데이터를 저장할 배열
			int[][] space = new int[100][100];
			// 저장한 데이터를 반시계로 90도 회전한 값을 저장할 배열
			int[][] space2 = new int[100][100];
			// 각 방향별 더한 값을 저장할 배열
			int[] sumSet = new int[202];

			// 테스트 케이스 번호 입력
			int testCaseNum = scan.nextInt();

			// 입력받은 데이터를 배열에 저장하는 반복문
			for (int i = 0; i < 100; i++) {
				for (int j = 0; j < 100; j++) {
					space[i][j] = scan.nextInt();
				}
			}

			// 주어진 데이터를 반시계로 회전하여 저장하는 반복문
			for (int i = 0; i < 100; i++) {
				for (int j = 0; j < 100; j++) {
					space2[i][j] = space[j][100 - i - 1];
				}
			}

			// 행 방향 합을 계산해 배열에 저장
			sumSet = dataSum(space, sumSet, 0);
			// 열 방향 합을 계산하 배열에 저장
			sumSet = dataSum(space2, sumSet, 100);

			// 좌상우하 대각선 합을 배열에 저장
			sumSet[200] = diagonalSum(space);
			// 우상좌하 대각선 합을 배열에 저장
			sumSet[201] = diagonalSum(space2);

			// 오름차순으로 정렬
			Arrays.sort(sumSet);

			// 출력 양식에 맞춰 출력
			System.out.println("#" + testCaseNum + " " + sumSet[201]);

		}

		// Scanner 종료
		scan.close();

	}

	// 행 방향 합을 구하는 메소드
	// subSet 배열에 저장할 인덱스를 고려하여 시작 인덱스도 매개변수로 받음
	static int[] dataSum(int[][] space, int[] sumSet, int startIndex) {

		for (int row = 0; row < 100; row++) {

			for (int column = 0; column < 100; column++) {
				sumSet[startIndex + row] += space[row][column];
			}

		}

		return sumSet;
	}

	// 대각선 합(좌상우하)를 구하는 메소드
	static int diagonalSum(int[][] space) {

		int dataSum = 0;

		for (int i = 0; i < space.length; i++) {
			dataSum += space[i][i];
		}

		return dataSum;
	}

}
