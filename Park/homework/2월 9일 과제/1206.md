import java.util.Scanner;

public class Problem1206 {

	// 1206. View
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 테스트 케이스별 결과값을 저장할 배열
		int[] views = new int[10];

		// 각 테스트 케이스별 결과값을 계산하는 반복문
		for (int i = 0; i < 10; i++) {

			// 공간의 크기 입력받아 저장
			int place = scan.nextInt();
			// 건물별 높이를 저장할 배열 생성
			int[] heights = new int[place];
			// 최고층 건물의 높이를 저장할 변수 생성
			int maxFloor = 0;

			// 건물별 높이를 배열에 저장하고
			// 최고층을 계산하는 반복문
			for (int j = 0; j < place; j++) {
				heights[j] = scan.nextInt(); // 입력한 높이를 배열에 저장

				// 최고층을 계산하는 조건문
				if (maxFloor < heights[j]) {
					maxFloor = heights[j];
				}
			}

			// 테스트 케이스의 결과값을 저장할 변수 생성
			int count = 0;

			// 1층부터 최고층까지 해당 층수에 조망권이 확보된 건물의 개수를 세는 반복문
			for (int j = 1; j <= maxFloor; j++) {

				// 해당 층에 건물이 존재하는지 여부를 저장하는 배열
				int[] floor = new int[place];

				// 해당 층보다 크거나 같은 높이의 건물이 있으면
				// 배열에 0대신 1을 저장하여 위치 표시
				for (int k = 0; k < place; k++) {

					if (heights[k] >= j) {
						floor[k] = 1;
					}

				}

				// 조망권의 조건을 만족하면 결과값에 개수 추가하는 반복문
				// 좌우측 2칸은 무조건 건물이 없어서
				// 인덱스 오류 없음
				for (int k = 0; k < place; k++) {

					// 조망권 확보 조건
					if ((floor[k] == 1) && (floor[k - 1] == 0) && (floor[k - 2] == 0) && (floor[k + 1] == 0)
							&& (floor[k + 2] == 0)) {
						count++;
					}
				}

			}

			// 해당 테스트 케이스의 결과를 배열에 저장
			views[i] = count;

		}

		// 출력 양식에 맞춰 테스트 케이스별 결과 출력
		for (int i = 0; i < 10; i++) {
			System.out.println("#" + (i + 1) + " " + views[i]);
		}

		scan.close();

	}

}
