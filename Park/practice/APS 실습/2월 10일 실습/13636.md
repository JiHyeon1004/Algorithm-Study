```java
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;

public class Problem13636 {

	// 13636. 연속된 1의 개수 중 최대값 구하기
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 테스트 케이스 개수 입력
		int testCase = scan.nextInt();
		// 테스트 케이스별 결과를 저장할 배열 생성
		int[] resultSet = new int[testCase];

		// 테스트 케이스별 연속 1 최대값 구하는 반복문
		for (int test = 0; test < testCase; test++) {
			// 데이터의 크기 입력
			int size = scan.nextInt();
			// 입력한 데이터를 한자리씩 배열에 저장
			String[] numStr = scan.next().split("");

			// 연속된 1의 개수를 저장할 리스트 생성
			List<Integer> countList = new ArrayList<>();

			// 초기 연속 1의 개수는 0
			int count = 0;

			// 연속된 1의 개수를 세는 반복문
			for (int i = 0; i < size; i++) {
				if (numStr[i].equals("1")) {
					count++; // 해당 자리가 1이면 count를 1 늘리고
					if (i == size - 1) {
						countList.add(count); // 마지막 자리가 1인 경우
					}
				} else if (numStr[i].equals("0")) {
					countList.add(count); // 해당 자리가 0이면 이전 자리까지 누적된 count를 리스트에 저장하고
					count = 0; // count는 0으로 초기화
				}
			}

			// 연속된 1의 개수가 저장된 리스트를
			// 오름차순으로 정렬하여
			Collections.sort(countList);

			// 마지막 인덱스가 최댓값이므로
			// 해당 인덱스의 값을 결과 저장 배열에 입력
			resultSet[test] = countList.get(countList.size() - 1);
		}

		// 출력 양식에 맞춰 출력
		for (int i = 0; i < testCase; i++) {
			System.out.println("#" + (i + 1) + " " + resultSet[i]);
		}

		// Scanner 종료
		scan.close();

	}

}
```
