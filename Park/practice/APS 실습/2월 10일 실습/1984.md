import java.util.Scanner;

public class Problem1984 {

	// 1984. 중간 평균값 구하기
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 테스트 케이스 입력
		int testCase = scan.nextInt();
		// 테스트 케이스별 결과 저장 배열 생성
		int[] resultSet = new int[testCase];

		// 테스트 케이스별 중간 평균값 구하는 반복문
		for (int test = 0; test < testCase; test++) {
			// 데이터의 최댓값을 저장할 변수 생성
			int maxValue = 0;
			// 데이터의 최솟값을 저장할 변수 생성
			int minValue = 10000;
			// 데이터의 총합을 저장할 변수 생성
			int dataSum = 0;

			// 10개의 데이터에 대해
			for (int i = 0; i < 10; i++) {
				// 데이터를 입력받아
				int data = scan.nextInt();

				// 최댓값인지 판별하고
				if (maxValue < data) {
					maxValue = data;
				}

				// 최솟값인지 판별하고
				if (minValue > data) {
					minValue = data;
				}

				// 데이터 총합에 입력한 값을 더함
				dataSum += data;
			}

			// 중간 평균 계산
			double dataAvg = (double) (dataSum - maxValue - minValue) / 8;

			// 반올림한 후 배열에 저장
			resultSet[test] = (int) Math.round(dataAvg);

		}

		// 출력 양식에 맞춰 결과 출력
		for (int i = 0; i < testCase; i++) {
			System.out.println("#" + (i + 1) + " " + resultSet[i]);
		}

		scan.close();

	}

}
