import java.util.Scanner;

public class Problem2063 {

	// 2063. 중간값 찾기
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 자료 개수를 저장할 변수
		int dataCount = scan.nextInt();
		// 입력한 자료를 저장할 배열
		int[] dataArr = new int[dataCount];

		// 입력한 자료를 배열에 저장
		for (int data = 0; data < dataCount; data++) {
			dataArr[data] = scan.nextInt();
		}

		// 선택 정렬로 오름차순으로 정렬
		for (int i = 0; i < dataCount - 1; i++) {

			int minIndex = i;

			for (int j = i + 1; j < dataCount; j++) {
				if (dataArr[minIndex] > dataArr[j]) {
					minIndex = j;
				}
			}

			int temp = dataArr[i];
			dataArr[i] = dataArr[minIndex];
			dataArr[minIndex] = temp;
		}

		// 결과값 출력
		System.out.println(dataArr[(dataCount) / 2]);

		// Scanner 종료
		scan.close();

	}

}
