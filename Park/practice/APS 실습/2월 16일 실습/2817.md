import java.util.Scanner;

public class Problem2817 {

	// 경우의 수를 저장할 변수
	static int count = 0;

	// 2817. 부분 수열의 합
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);
		// 테스트 케이스 개수 입력
		int testCaseNum = scan.nextInt();

		// 테스트 케이스 개수만큼 반복
		for (int test = 1; test <= testCaseNum; test++) {

			// 수열 크기
			int N = scan.nextInt();
			// 합의 결과 조건
			int K = scan.nextInt();

			// 해당 숫자 선택 여부를 저장할 배열
			int[] choice = new int[N];
			// 수열 데이터를 저장할 배열
			int[] numArr = new int[N];

			// 배열에 수열 데이터 저장
			for (int i = 0; i < N; i++) {
				numArr[i] = scan.nextInt();
			}

			// 부분집합 메소드 실행
			powerSet(0, choice, numArr, K);

			// 부분집합 메소드를 통해 센 부분집합의 개수 출력
			System.out.println("#" + test + " " + count);
			// 부분집합의 개수 변수 초기화
			count = 0;

		}
		
		scan.close();

	}

	// 부분집합을 만들고
	// 합을 계산하고
	// 원하는 값과 같으면 count에 1을 추가하는 메소드
	public static void powerSet(int index, int[] choice, int[] numArr, int K) {

		// 기본
		if (index == choice.length) {

			// 부분집합의 원소의 합을 저장할 변수
			int sum = 0;

			// 지금 만든 부분집합의 원소의 합을 구하는 반복문
			for (int i = 0; i < choice.length; i++) {
				if (choice[i] == 1) {
					sum += numArr[i];
				}

			}
			// 원하는 값과 같으면 count에 1 추가
			if (sum == K) {
				count++;
			}

			return;
		}

		// 재귀
		choice[index] = 1;
		powerSet(index + 1, choice, numArr, K);

		choice[index] = 0;
		powerSet(index + 1, choice, numArr, K);

	}

}
