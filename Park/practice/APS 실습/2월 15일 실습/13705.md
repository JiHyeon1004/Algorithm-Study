import java.util.Scanner;

public class Problem13705 {

	// 13705. 파동파동
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 테스트 케이스 개수 입력
		int testCaseNum = scan.nextInt();

		// 입력한 개수만큼 반복
		for (int test = 1; test <= testCaseNum; test++) {

			// 크기 입력
			int size = scan.nextInt();
			// 시작 값 입력
			int startValue = scan.nextInt();
			// 시작 행
			int startRow = scan.nextInt() - 1;
			// 시작 열
			int startCol = scan.nextInt() - 1;
			// 증감
			int change = scan.nextInt();

			// 파동 수치를 저장할 배열
			int[][] space = new int[size][size];
			// 시작 위치 파동 값 입력
			space[startRow][startCol] = startValue;

			// 해당 위치의 파동값을 구하는 반복문
			for (int i = 0; i < size; i++) {
				for (int j = 0; j < size; j++) {
					// 행 거리와 열 거리 중 더 큰 값이 거리
					int distance = Math.max(Math.abs(i - startRow), Math.abs(j - startCol));
					// 거리만큼 증감값에 곱한 후 초기값과 더한 것이 해당 위치의 파동값
					int value = startValue + distance * change;
					if (value > 0) {
						space[i][j] = Math.min(value, 255); // 255보다 커지면 255를 입력
					}
				}
			}

			// 출력 양식
			System.out.print("#" + test + " ");

			// 결과 출력 반복문
			for (int i = 0; i < size; i++) {
				int sum = 0;
				for (int j = 0; j < size; j++) {
					sum += space[i][j];
				}
				System.out.print(sum + " ");
			}

			System.out.println(); // 줄 넘김

		}

		// Scanner 종료
		scan.close();

	}

}
