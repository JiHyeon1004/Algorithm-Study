import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Problem13707 {

	// 13707. 기지국
	public static void main(String[] args) throws NumberFormatException, IOException {

		// 입력 스트림
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

		// 테스트 케이스 개수 입력
		int testCaseNum = Integer.parseInt(br.readLine());

		// 입력한 개수만큼 반복
		for (int test = 1; test <= testCaseNum; test++) {

			// 공간 크기 입력
			int size = Integer.parseInt(br.readLine());
			// 공간 정보를 저장할 배열
			char[][] space = new char[size][size];

			// 배열에 공간정보 입력 반복문
			for (int i = 0; i < size; i++) {
				String str = br.readLine();
				space[i] = str.toCharArray();
			}

			// 기지국의 종류에 맞춰
			// 해당 기지국이 커버 가능한 곳의 위치 정보를 X로 바꾸는 메소드 활용
			for (int i = 0; i < size; i++) {
				for (int j = 0; j < size; j++) {
					if (space[i][j] != 'X' && space[i][j] != 'H') {
						tower(space, i, j, space[i][j]);
					}
				}
			}

			// 남은 집의 개수를 저장할 변수
			int count = 0;

			// 남은 집의 개수를 세는 반복문
			for (int i = 0; i < size; i++) {
				for (int j = 0; j < size; j++) {
					if (space[i][j] == 'H') {
						count++;
					}
				}
			}

			// 결과 출력
			System.out.println("#" + test + " " + count);

		}

	}

	// 기지국이 커버 가능한 범위를 표시하는 메소드
	public static char[][] tower(char[][] space, int row, int col, char tower) {

		for (int i = 1; i <= tower - 'A' + 1; i++) {

			if (col - i >= 0) {
				space[row][col - i] = 'X';
			}

			if (col + i < space.length) {
				space[row][col + i] = 'X';
			}

			if (row - i >= 0) {
				space[row - i][col] = 'X';
			}

			if (row + i < space.length) {
				space[row + i][col] = 'X';
			}

		}

		return space;
	}

}
