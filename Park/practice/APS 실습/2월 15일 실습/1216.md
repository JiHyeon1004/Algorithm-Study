import java.util.Scanner;

public class Problem1216 {

	// 1216. 회문2
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 10번의 테스트 케이스 반복
		for (int test = 0; test < 10; test++) {

			// 테스트 케이스 번호 입력
			int testCase = scan.nextInt();
			// 문자 정보 저장 배열
			char[][] space = new char[100][100];
			// 배열을 시계 방향으로 회전한 배열
			char[][] space2 = new char[100][100];
			// 최장 회문 길이 저장 변수
			int maxLength = 0;

			// 배열에 데이터 입력
			for (int i = 0; i < 100; i++) {
				String str = scan.next();
				space[i] = str.toCharArray();
			}

			// 배열 회전
			for (int i = 0; i < 100; i++) {
				for (int j = 0; j < 100; j++) {
					space2[i][j] = space[j][i];
				}
			}

			// 최장 길이 계산하는 반복문
			for (int length = 100; length > 0; length--) {
				if (checkWord(space, length) || checkWord(space2, length)) {
					maxLength = length;
					break;
				}
			}

			// 결과 출력
			System.out.println("#" + testCase + " " + maxLength);

		}

		scan.close();

	}

	// 해당 길이의 회문이 있는지 확인하는 반복문
	public static boolean checkWord(char[][] space, int length) {

		// 존재여부 저장 변수
		boolean isExist = false;

		// 회문인지 판별하는 반복문
		for (int i = 0; i < 100; i++) {
			for (int j = 0; j < 100 - length + 1; j++) {
				StringBuilder sb = new StringBuilder();

				for (int k = 0; k < length; k++) {
					sb.append(space[i][j + k]);
				}

				String str1 = sb.toString();
				sb.reverse();
				String str2 = sb.toString();

				if (str1.equals(str2)) {
					isExist = true;
				}
			}
		}

		// 존재 여부 반환
		return isExist;
	}

}
