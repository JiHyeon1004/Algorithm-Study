import java.util.Scanner;

public class Problem1966 {

	// 1966. 숫자를 정렬하자
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 테스트 케이스 개수 입력
		int testCaseSize = scan.nextInt();

		// 테스트 케이스별 결과 출력하는 반복문
		for (int testCase = 1; testCase <= testCaseSize; testCase++) {
			// 테스트 케이스의 입력되는 숫자의 개수
			int dataSize = scan.nextInt();
			// 데이터를 저장할 배열
			int[] data = new int[dataSize];
			// 입력된 데이터의 최댓값을 저장할 변수
			int max = Integer.MIN_VALUE;
			// 입력된 데이터의 최솟값을 저장할 변수
			int min = Integer.MAX_VALUE;

			// 배열에 데이터를 저장하고
			// 입력된 데이터의 최대, 최소를 찾는 반복문
			for (int i = 0; i < dataSize; i++) {
				// 배열에 데이터 저장
				data[i] = scan.nextInt();

				// 최댓값 찾는 조건문
				if (max < data[i]) {
					max = data[i];
				}

				// 최솟값 찾는 조건문
				if (min > data[i]) {
					min = data[i];
				}
			}

			// 최솟값이 양수면 0을 최솟값으로
			// 음수면 해당 값을 최솟값으로 취함
			min = Math.min(0, min);

			// 데이터별 출현 횟수를 저장할 배열 생성
			int[] count = new int[max + min + 1];

			// 데이터별 해당 인덱스에 횟수 추가 반복문
			for (int i = 0; i < dataSize; i++) {
				count[data[i] + min]++;
			}

			// 누적 합
			for (int i = 1; i < count.length; i++) {
				count[i] = count[i] + count[i - 1];
			}

			// 정렬된 데이터를 저장할 배열
			int[] temp = new int[dataSize];

			// 누적 합 배열을 활용하여 데이터 정렬
			for (int i = dataSize - 1; i >= 0; i--) {
				temp[count[data[i] + min] - 1] = data[i];
				count[data[i] + min]--;
			}

			// 출력 양식
			System.out.print("#" + testCase + " ");

			// 테스트 케이스 결과 출력 반복문
			for (int i = 0; i < temp.length; i++) {
				System.out.print(temp[i] + " ");
			}

			// 한 줄 간격
			System.out.println();

		}

		// Scanner 종료
		scan.close();

	}

}
