import java.util.Scanner;

public class Problem1208 {

	// 1208. Flatten
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 10번의 테스트 케이스 당 결과 값을 출력하는 반복문
		for (int testCase = 0; testCase < 10; testCase++) {

			// 테스트 케이스의 덤프 제한 횟수 입력
			int dump = scan.nextInt();

			// 박스별 높이를 저장할 배열
			int[] place = new int[100];

			// 배열에 높이 데이터 입력하는 반복문
			for (int i = 0; i < 100; i++) {
				place[i] = scan.nextInt();
			}

			// 초기 최대 높이와 최저 높이
			int maxHeight = 0;
			int minHeight = 100;

			// 제한된 횟수만큼 덤프를 진행하는 반복문
			for (int i = 0; i <= dump; i++) {
				maxHeight = findMaxHeight(place); // 해당 인덱스 횟수만큼 덤프를 진행한 후 최대 높이
				minHeight = findMinHeight(place); // 해당 인덱스 횟수만큼 덤프를 진행한 후 최소 높이

				// 평탄화가 완료되면 반복문 종료
				if (maxHeight - minHeight <= 1) {
					break;
				}
			}
			
			// 출력 양식에 맞춰 출력
			System.out.println("#" + (testCase + 1) + " " + (maxHeight - minHeight));

		}
		
		// Scanner 종료
		scan.close();

	}

	// 현재 최대 높이를 구하고 덤프를 수행하는 메소드
	static int findMaxHeight(int[] place) {

		// 최대 박스가 쌓여 있는 곳의 위치
		int maxHeightIndex = 0;
		// 최대 박스 높이
		int maxHeight = 0;

		// 최대 박스가 쌓인 곳의 위치를 찾는 반복문
		for (int i = 0; i < place.length; i++) {
			if (place[maxHeightIndex] < place[i]) {
				maxHeightIndex = i;
			}
		}

		// 해당 인덱스의 높이 값을 반환하고
		maxHeight = place[maxHeightIndex];
		// 덤프 수행
		place[maxHeightIndex]--;

		return maxHeight;

	}

	// 현재 최저 높이를 구하고 덤프를 수행하는 메소드
	static int findMinHeight(int[] place) {

		// 최소 박스가 쌓여 있는 곳의 위치
		int minHeightIndex = 0;
		// 최소 박스 높이
		int minHeight = 100;

		// 최소 박스가 쌓인 곳의 위치를 찾는 반복문
		for (int i = 0; i < place.length; i++) {
			if (place[minHeightIndex] > place[i]) {
				minHeightIndex = i;
			}
		}

		// 해당 인덱스의 높이 값을 반환하고
		minHeight = place[minHeightIndex];
		// 덤프 수행
		place[minHeightIndex]++;

		return minHeight;

	}

}
