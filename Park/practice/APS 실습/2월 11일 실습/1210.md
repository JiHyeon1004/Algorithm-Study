import java.util.Scanner;

public class Problem1210 {

	// 1210. Ladder1
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 10번의 테스트 케이스 반복
		for (int test = 0; test < 10; test++) {
			// 테스트 케이스 번호 입력
			int testCaseNum = scan.nextInt();

			// 사다리 정보를 저장할 배열 생성
			int[][] space = new int[100][100];

			// 배열에 사다리 정보 저장
			for (int row = 0; row < 100; row++) {
				for (int column = 0; column < 100; column++) {
					space[row][column] = scan.nextInt();
				}
			}

			// 델타 탐색
			// 좌, 우, 하 순
			int[] dr = { 0, 0, 1 };
			int[] dc = { -1, 1, 0 };

			// 우리가 찾으려는 것은 열 번호이므로
			// 행 번호에 대한 반복문 X
			for (int column = 0; column < 100; column++) {
				// 행 번호 저장 변수
				// 첫 줄에서 시작하므로 x는 0으로 초기화
				int x = 0;
				// 열 번호 저장 변수
				int y = column;

				// 시작점부터 0이면 반복 미실행
				if (space[x][y] == 0) {
					continue;
				}

				// 맨 밑 칸에 도착할 때 까지 반복
				while (x < 99) {

					// 현재 좌표 기준 델타 탐색을 실행한 후
					for (int k = 0; k < 3; k++) {
						int newX = x + dr[k];
						int newY = y + dc[k];

						// 새로운 좌표가 범위 밖이면 다음 방향으로 탐색을 하고
						if (newX < 0 || newX >= 100 || newY < 0 || newY >= 100) {
							continue;
						}

						// 새로운 좌표의 방향이 좌 또는 우면서 길이 있으면
						if (space[newX][newY] == 1 && k != 2) {
							// 해당 방향의 끝까지 이동하는 반복문
							while (space[newX][newY] == 1) {
								x = newX;
								y = newY;

								// 방향 유지를 위해 k 인덱스 사용
								newX = x + dr[k];
								newY = y + dc[k];

								// 다음 좌표가 범위 밖이거나
								// 길이 없으면 while 문을 종료하고
								if (newX < 0 || newX >= 100 || newY < 0 || newY >= 100) {
									break;
								}
							}

							// 밑으로 한 칸 이동한 후
							x++;
							// 반복문 종료 후 다시 탐색 실행
							break;
						} else if ((space[newX][newY] == 1 || space[newX][newY] == 2) && k == 2) {
							x++; // 좌우에 길이 없으면 1칸 내려간 후
							// 다시 탐색 실행
							break;
						}
					}
				}

				// 마지막으로 도착한 지점이 2로 정답이면
				// 결과를 출력하고 반복문 종료
				if (space[x][y] == 2) {
					System.out.println("#" + testCaseNum + " " + column);
					break;
				}

			}
		}

		// Scanner 종료
		scan.close();

	}

}
