```java
import java.util.Scanner;

public class Problem1966 {

	// 1966. 숫자를 정렬하자
	public static void main(String[] args) {

		// Scanner 생성
		Scanner scan = new Scanner(System.in);

		// 테스트 케이스 입력받아 저장
		int testCase = scan.nextInt();

		// 테스트 케이스별 정렬된 배열을 저장할 배열 생성
		int[][] sortArr = new int[testCase][];

		// 입력된 값을 배열에 저장하고
		// 배열을 정렬하고
		// 정렬한 배열을 앞서 생성한 배열에 입력
		for (int i = 0; i < testCase; i++) {

			// 자료 개수 입력받아 저장
			int size = scan.nextInt();

			// 자료 개수에 맞춰 배열 생성
			int[] arr = new int[size];

			// 입력받은 값을 차례대로 배열에 저장
			for (int j = 0; j < size; j++) {
				arr[j] = scan.nextInt();
			}

			// 버블 정렬로 배열 정렬
			for (int j = size - 1; j > 0; j--) {

				for (int k = 0; k < j; k++) {

					if (arr[k] > arr[k + 1]) {

						int temp = arr[k];
						arr[k] = arr[k + 1];
						arr[k + 1] = temp;

					}

				}

			}

			// 정렬한 배열을 배열 저장 배열에 입력
			sortArr[i] = arr;
		}

		// 테스트 케이스별 정렬 결과를 출력 양식에 맞춰 출력
		for (int i = 0; i < testCase; i++) {

			System.out.print("#" + (i + 1) + " ");

			for (int j = 0; j < sortArr[i].length; j++) {
				System.out.print(sortArr[i][j] + " ");
			}
			
			System.out.println();
		}

		scan.close();

	}

}
```
