```java
public static void main(String[] args) {

		Scanner scan = new Scanner(System.in);

		int studentNumber; // 조회하려는 학생 수
		int month; // 태어난 달
		
		System.out.println("[ 내가 태어난 계절은? ]");
		System.out.print("계절을 찾으려는 학생 수를 입력하세요. >> ");
		
		studentNumber = scan.nextInt();
		
		int i = 0;
		
		while (i < studentNumber) {

			System.out.print("학생 " + (i + 1) + "이(가) 태어난 달을 입력하세요 >> ");

			month = scan.nextInt();

			switch (month) {
			case 12:
			case 1:
			case 2:
				System.out.println("학생 " + (i + 1) + "이(가) 태어난 달은 겨울입니다.");
				System.out.println("");
				i++;
				break;
			case 3:
			case 4:
			case 5:
				System.out.println("학생 " + (i + 1) + "이(가) 태어난 달은 봄입니다.");
				System.out.println("");
				i++;
				break;
			case 6:
			case 7:
			case 8:
				System.out.println("학생 " + (i + 1) + "이(가) 태어난 달은 여름입니다.");
				System.out.println("");
				i++;
				break;
			case 9:
			case 10:
			case 11:
				System.out.println("학생 " + (i + 1) + "이(가) 태어난 달은 가을입니다.");
				System.out.println("");
				i++;
				break;
			default:
				System.out.println("0과 12 사이의 숫자만 입력하세요.");
				System.out.println("");
			}

		}

		System.out.println("계절 확인 프로그램이 종료되었습니다.");

		scan.close();

	}
```
